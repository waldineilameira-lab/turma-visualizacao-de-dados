import pandas as pd

# ── LEITURA ───────────────────────────────────────────────────
df = pd.read_csv("base_rh.csv", sep=";", encoding="cp1252", decimal=",")

cola rápida — Dia 02 · regexPython
import re

# ── MÓDULO RE ─────────────────────────────────────────────────
re.sub(padrao, substituto, texto)    # substitui o padrão pelo substituto
re.search(padrao, texto)             # procura o padrão → retorna Match ou None
re.findall(padrao, texto)            # retorna lista de todas as ocorrências

# ── PANDAS COM TEXTO ──────────────────────────────────────────
df["col"].str.contains("padrão", regex=True)   # retorna True/False por linha
df["col"].str.extract(r"(grupo)")             # extrai o grupo capturado
df["col"].str.replace("padrão", "", regex=True) # substitui direto na coluna

# ── METACARACTERES MAIS USADOS ────────────────────────────────
# ^     → início da string
# \s    → espaço em branco
# \d    → qualquer dígito (0–9)
# \w    → letra, número ou underscore
# +     → 1 ou mais repetições
# *     → 0 ou mais repetições
# \.    → ponto literal (o ponto sem \ = qualquer caractere)
# (abc) → grupo de captura
# (?:abc) → grupo sem captura (só agrupa alternativas)

Desafio 4 — Limpeza de nomes e sobrenomes mais comuns
gabarito · desafio 4Python
import re

# Contar com str.contains
padrao_titulo = r"^(Sr\.|Sra\.|Dr\.|Dra\.|Srta\.)"
com_titulo = df["Nome"].str.contains(padrao_titulo, regex=True).sum()
print(f"Nomes com título: {com_titulo}")   # → 175

# Remover títulos com re.sub dentro de .apply()
def remover_titulo(nome):
    if not isinstance(nome, str):
        return nome
    return re.sub(r"^(?:Sr\.|Sra\.|Dr\.|Dra\.|Srta\.)\s+", "", nome).strip()

df_limpo["Nome_Limpo"] = df_limpo["Nome"].apply(remover_titulo)

# Mostrar 5 exemplos antes × depois
mask = df["Nome"].str.contains(padrao_titulo, regex=True)
print("\n5 exemplos antes × depois:")
for i in df[mask].head(5).index:
    print(f"  '{df.loc[i,'Nome']}' → '{df_limpo.loc[i,'Nome_Limpo']}'")

# Extrair sobrenome com str.extract
# r"(\w+)$" → captura 1+ letras/números no FIM da string
df_limpo["Sobrenome"] = df_limpo["Nome_Limpo"].str.extract(r"(\w+)$")

print("\n5 sobrenomes mais comuns:")
print(df_limpo["Sobrenome"].value_counts().head(5))

Desafio 6 · Pergunta 3 — Salário médio com vs. sem título
gabarito · desafio 6 · pergunta 3Python
# Criamos a máscara no df ORIGINAL (antes da limpeza)
mask_titulo = df["Nome"].str.contains(
    r"^(Sr\.|Sra\.|Dr\.|Dra\.|Srta\.)", regex=True
)

sal_com_titulo    = df[mask_titulo]["Salario"].mean()
sal_sem_titulo    = df[~mask_titulo]["Salario"].mean()
# ~mask_titulo → inverte a máscara (~ = NOT lógico)

print(f"Funcionários COM título : {mask_titulo.sum()}")
print(f"  Salário médio         : R$ {sal_com_titulo:,.2f}")
print(f"\nFuncionários SEM título : {(~mask_titulo).sum()}")
print(f"  Salário médio         : R$ {sal_sem_titulo:,.2f}")
print(f"\nDiferença               : R$ {abs(sal_com_titulo - sal_sem_titulo):,.2f}")
